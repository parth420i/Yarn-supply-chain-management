    <script>
      async function main() {
        const fetchFile = async (url) => {
          const res = await fetch(url);
          if (!res.ok) throw new Error(`Failed to fetch ${url}`);
          return await res.text();
        };

        try {
          // Fetch files from the current directory
          const [appPy, mainPy, csvData] = await Promise.all([
            fetchFile("./app.py"),
            fetchFile("./main.py"),
            fetchFile("./yarn_supplychain_surat.csv")
          ]);

          stlite.mount(
            {
              requirements: ["pandas", "numpy", "scikit-learn", "plotly"],
              entrypoint: "app.py",
              files: {
                "app.py": appPy,
                "main.py": mainPy,
                "yarn_supplychain_surat.csv": csvData
              },
            },
            document.getElementById("stlite")
          );
        } catch (err) {
          console.error(err);
          document.getElementById("stlite").innerHTML = `<div style="padding: 20px; color: red;">Error loading application: ${err.message}</div>`;
        }
      }
      main();
    </script>
